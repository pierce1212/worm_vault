const CONFIG = {
  taskSourceFiles: ["工作任务.md"],
  detailFile: "任务详情.md",
  dashboardFile: "obsidian_tasks_command_center.md",
  reviewIntervals: [1, 2, 4, 7, 15, 30],
  longTermTags: ["#长期", "#longterm", "#someday", "#长期任务"],
  projectTags: ["#project", "#项目", "#工作", "#学习", "#通信", "#安全", "#生活", "#机器人", "#嵌入式"]
};

const { DateTime } = dv.luxon;
const today = DateTime.now().startOf("day");
const sourceFile = dv.current().file.path;
const sourceFolder = sourceFile.includes("/") ? sourceFile.slice(0, sourceFile.lastIndexOf("/")) : "";
const localPath = (name) => sourceFolder && !name.startsWith(`${sourceFolder}/`) ? `${sourceFolder}/${name}` : name;
const taskSourcePaths = CONFIG.taskSourceFiles.map(localPath);
const detailPath = localPath(CONFIG.detailFile);
const dashboardPath = localPath(CONFIG.dashboardFile);

function arr(value) {
  if (!value) return [];
  if (Array.isArray(value)) return value;
  if (typeof value.array === "function") return value.array();
  try { return Array.from(value); } catch { return []; }
}

function esc(value) {
  return String(value ?? "")
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;");
}

function mount(markup) {
  const wrap = document.createElement("div");
  wrap.innerHTML = markup.trim();
  const first = wrap.firstElementChild;
  if (first) dv.container.appendChild(first);
  return first;
}

function dt(value) {
  if (!value) return null;
  if (value.isLuxonDateTime || (value.toISO && value.ts)) return value;
  const raw = String(value).trim();
  if (!raw) return null;
  const plain = raw.replace(/^\[\[/, "").replace(/\]\]$/, "").replace(" ", "T");
  let parsed = DateTime.fromISO(plain);
  if (!parsed.isValid) parsed = DateTime.fromISO(plain.replace(/\//g, "-"));
  if (!parsed.isValid) parsed = DateTime.fromFormat(raw, "yyyy-LL-dd HH:mm");
  if (!parsed.isValid) parsed = DateTime.fromFormat(raw, "yyyy-LL-dd");
  if (!parsed.isValid) parsed = DateTime.fromFormat(raw, "yyyy/MM/dd HH:mm");
  if (!parsed.isValid) parsed = DateTime.fromFormat(raw, "yyyy/MM/dd");
  return parsed.isValid ? parsed : null;
}

function date(value) {
  const parsed = dt(value);
  return parsed ? parsed.startOf("day") : null;
}

function sameDay(a, b) {
  return a && b && a.hasSame(b, "day");
}

function fmtDate(value, format = "yyyy-LL-dd") {
  const parsed = date(value);
  return parsed ? parsed.toFormat(format) : "-";
}

function taskTags(task) {
  return arr(task.tags).map(String);
}

function completedAtValue(task) {
  return task.completedAt ?? task["completedAt"] ?? task["完成时间"] ?? task.completion;
}

function symbolDay(task, symbol) {
  const raw = String(task.text ?? "");
  const match = raw.match(new RegExp(`${symbol}\\s*(\\d{4}-\\d{2}-\\d{2})`, "u"));
  return match ? date(match[1]) : null;
}

function completedDay(task) {
  return date(completedAtValue(task)) ?? symbolDay(task, "✅");
}

function inlineField(text, key) {
  const pattern = new RegExp(`\\[(?:${key})::\\s*([^\\]]+)\\]`, "iu");
  const match = String(text ?? "").match(pattern);
  return match ? match[1].trim() : "";
}

function childListItems(task) {
  const output = [];
  function walk(items) {
    for (const item of arr(items)) {
      output.push(item);
      walk(item.children);
    }
  }
  walk(task.children);
  return output;
}

function cleanText(text, removeTags = true) {
  let output = String(text ?? "")
    .replace(/[🔺⏫🔼🔽⏬]\s*/gu, "")
    .replace(/(?:📅|⏳|🛫|➕|✅|❌)\s*\d{4}-\d{2}-\d{2}/gu, "")
    .replace(/🔁\s*[^#\[]+/gu, "")
    .replace(/\[(?:completedAt|完成时间|needsReview|需要复习|reviewEnabled|复习开关)::[^\]]+\]/giu, "")
    .replace(/\s+/g, " ")
    .trim();

  if (removeTags) {
    for (const tag of [...taskTagUniverse, ...CONFIG.projectTags, ...CONFIG.longTermTags]) {
      output = output.replaceAll(tag, "");
    }
    output = output.replace(/\s+/g, " ").trim();
  }
  return output;
}

function sourcePath(task) {
  return task.path ?? taskSourcePaths[0];
}

function taskId(task) {
  return `${sourcePath(task)}:${task.line ?? ""}:${cleanText(task.text)}`;
}

function openPath(path) {
  app.workspace.openLinkText(path, sourceFile, false);
}

function reviewRecords(task) {
  return childListItems(task)
    .map(item => {
      const text = String(item.text ?? "").trim();
      const due = date(inlineField(text, "review|复习"));
      if (!due) return null;
      return {
        due,
        interval: Number(inlineField(text, "interval|间隔")) || null,
        done: date(inlineField(text, "review_done|复习完成|done|完成")),
        text
      };
    })
    .filter(Boolean);
}

function reviewChoice(task) {
  const raw = task.needsReview ?? task.reviewEnabled ?? task["需要复习"] ?? task["复习开关"] ?? inlineField(task.text, "needsReview|需要复习|reviewEnabled|复习开关");
  if (raw === null || raw === undefined || raw === "") return null;
  const value = String(raw).trim().toLowerCase();
  if (["1", "true", "yes", "y", "on", "是", "是的", "需要", "需要复习", "复习", "开启", "打开"].includes(value)) return true;
  if (["0", "false", "no", "n", "off", "否", "不用", "不用复习", "不需要", "不需要复习", "无需", "无需复习", "不复习", "关闭", "关"].includes(value)) return false;
  return null;
}

function shouldReviewTask(task, records = null) {
  const choice = reviewChoice(task);
  if (choice !== null) return choice;
  return (records ?? reviewRecords(task)).length > 0;
}

function reviewPlan(task) {
  const doneDay = completedDay(task);
  const records = reviewRecords(task);
  if (!doneDay || !shouldReviewTask(task, records)) return [];
  return CONFIG.reviewIntervals.map(interval => {
    const due = doneDay.plus({ days: interval });
    const record = records.find(item => item.interval === interval || sameDay(item.due, due));
    const done = record?.done ?? null;
    const state = done ? "done" : due < today ? "overdue" : sameDay(due, today) ? "today" : "future";
    return { task, interval, due, done, state, doneDay };
  });
}

function stateText(item) {
  if (item.state === "done") return `已复习 ${fmtDate(item.done)}`;
  if (item.state === "overdue") return `逾期 ${Math.max(0, Math.floor(today.diff(item.due, "days").days))} 天`;
  if (item.state === "today") return "今天复习";
  return `${Math.max(0, Math.floor(item.due.diff(today, "days").days))} 天后`;
}

function tagsHtml(task) {
  return taskTags(task)
    .filter(tag => !CONFIG.longTermTags.includes(tag))
    .slice(0, 4)
    .map(tag => `<span>${esc(tag)}</span>`)
    .join("");
}

function nextReview(plan) {
  return plan.find(item => item.state === "overdue" || item.state === "today" || item.state === "future") ?? plan.at(-1);
}

function trackProgress(plan) {
  return Math.round(plan.filter(item => item.state === "done").length / Math.max(1, plan.length) * 100);
}

function reviewDot(item, index) {
  return `
    <button class="tcc-review-dot ${item.state}" type="button" data-detail-task="${esc(taskId(item.task))}" style="--i:${index}">
      <span>D+${item.interval}</span>
      <strong>${fmtDate(item.due, "LL-dd")}</strong>
      <em>${esc(stateText(item))}</em>
    </button>
  `;
}

function reviewTrackCard(group, index) {
  const { task, plan } = group;
  const next = nextReview(plan);
  const progress = trackProgress(plan);
  const dueCount = plan.filter(item => item.state === "today" || item.state === "overdue").length;
  return `
    <article class="tcc-review-track-card ${dueCount ? "needs-action" : ""}" data-detail-task="${esc(taskId(task))}" style="--card-index:${index}">
      <header>
        <div>
          <span class="tcc-review-kicker">完成日 ${fmtDate(completedDay(task))}</span>
          <h3>${esc(cleanText(task.text))}</h3>
          <div class="tcc-task-tags">${tagsHtml(task)}</div>
        </div>
        <div class="tcc-review-orb ${next?.state ?? "done"}">
          <strong>${next ? `D+${next.interval}` : "Done"}</strong>
          <span>${next ? stateText(next) : "已完成"}</span>
        </div>
      </header>

      <div class="tcc-review-rail" style="--progress:${progress}%">
        ${plan.map(reviewDot).join("")}
      </div>

      <footer>
        <span>${progress}% 已复习</span>
        <em>${next ? `下一节点 ${fmtDate(next.due)} · ${stateText(next)}` : "复习计划已完成"}</em>
      </footer>
    </article>
  `;
}

const sourcePages = dv.pages().where(page => taskSourcePaths.includes(page.file.path));
const tasks = sourcePages.file.tasks.where(task => task.task).array();
const taskTagUniverse = [...new Set(tasks.flatMap(task => taskTags(task)))];
const reviewGroups = tasks
  .filter(task => task.completed && completedDay(task))
  .map(task => ({ task, plan: reviewPlan(task) }))
  .filter(group => group.plan.length)
  .sort((a, b) => {
    const an = nextReview(a.plan);
    const bn = nextReview(b.plan);
    const aUrgent = a.plan.some(item => item.state === "today" || item.state === "overdue") ? 0 : 1;
    const bUrgent = b.plan.some(item => item.state === "today" || item.state === "overdue") ? 0 : 1;
    if (aUrgent !== bUrgent) return aUrgent - bUrgent;
    return (an?.due.toMillis() ?? 0) - (bn?.due.toMillis() ?? 0);
  });

const allItems = reviewGroups.flatMap(group => group.plan);
const dueItems = allItems.filter(item => item.state === "today" || item.state === "overdue");
const futureItems = allItems.filter(item => item.state === "future");
const doneItems = allItems.filter(item => item.state === "done");

const root = mount(`
  <section class="tcc-shell tcc-review-timeline-shell">
    <section class="tcc-review-hero tcc-review-visual-hero">
      <div>
        <p class="tcc-eyebrow">${today.toFormat("yyyy-LL-dd")} · 艾宾浩斯复习轨道</p>
        <h2>复习星轨</h2>
        <p>每个任务一条复习轨道，节点按 D+1 / D+2 / D+4 / D+7 / D+15 / D+30 展开。</p>
      </div>
      <div class="tcc-detail-actions">
        <button type="button" data-open-path="${esc(dashboardPath)}">返回驾驶舱</button>
      </div>
    </section>

    <section class="tcc-review-stats tcc-review-visual-stats">
      <article><span>待处理</span><strong>${dueItems.length}</strong><em>今天和逾期节点</em></article>
      <article><span>未来计划</span><strong>${futureItems.length}</strong><em>尚未到期节点</em></article>
      <article><span>已复习</span><strong>${doneItems.length}</strong><em>已记录节点</em></article>
    </section>

    <section class="tcc-review-track-grid">
      ${reviewGroups.map(reviewTrackCard).join("") || `<p class="tcc-empty">暂无开启复习的完成任务。</p>`}
    </section>
  </section>
`);

root.querySelectorAll("[data-open-path]").forEach(element => {
  element.addEventListener("click", () => openPath(element.dataset.openPath));
});

root.querySelectorAll("[data-detail-task]").forEach(element => {
  element.addEventListener("click", event => {
    event.stopPropagation();
    window.localStorage.setItem("tcc:selectedTaskId", element.dataset.detailTask);
    window.localStorage.setItem("tcc:selectedTaskAt", new Date().toISOString());
    openPath(detailPath);
  });
});
