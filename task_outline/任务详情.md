function dt(value) {
  if (!value) return null;
  if (value.isLuxonDateTime || (value.toISO && value.ts)) return value;

  if (!raw) return null;
  const plain = raw.replace(/^\[\[/, "").replace(/\]\]$/, "").replace(" ", "T");

}

function hasClockTime(value) {

function asTimelineMoment(value, page = null) {
  const parsed = dt(value);
  if (!parsed) return null;
  if (hasClockTime(value)) return parsed.set({ second: 0, millisecond: 0 });

  const fallback = page ? dt(page.file?.mtime ?? page.file?.ctime) : null;
  if (fallback) {
    return parsed.set({
      hour: fallback.hour,
      minute: fallback.minute,
      second: 0,
      millisecond: 0
    });
  }

  return parsed.startOf("day");
}

async function addTimelineNote(task, payload) {
  const path = sourcePath(task);
  const file = app.vault.getAbstractFileByPath(path);
  const line = lineNumber(task);
  if (!file || line < 0) return false;

  const text = String(payload.text ?? "").trim();
  if (!text) return false;

  const content = await app.vault.read(file);
  const lines = content.split(/\r?\n/);
  const sourceLine = lines[line] ?? "";
  const indent = (sourceLine.match(/^\s*/) ?? [""])[0] + "  ";
  const moment = asTimelineMoment(payload.moment ?? payload.day) ?? now.set({ second: 0, millisecond: 0 });
  const type = String(payload.type ?? "说明").trim() || "说明";
  const progress = progressValue(payload.progress);
  const progressField = progress !== null ? ` [progress:: ${progress}%]` : "";
  const noteLine = `${indent}- [timeline:: ${moment.toFormat("yyyy-LL-dd HH:mm")}] [type:: ${type}]${progressField} ${text}`;

  let insertAt = line + 1;
  while (insertAt < lines.length) {
    const next = lines[insertAt];
    if (!next.trim()) {
      insertAt++;
      continue;
    }
    const nextIndent = (next.match(/^\s*/) ?? [""])[0].length;
    if (nextIndent <= (sourceLine.match(/^\s*/) ?? [""])[0].length && /^\s*[-*]\s+/.test(next)) break;
    insertAt++;
  }

  lines.splice(insertAt, 0, noteLine);
  await app.vault.modify(file, lines.join("\n"));
  return true;
}

function completionProgress(task) {

function timeProgress(task) {

  const due = dueDay(task);

  const elapsed = today.diff(start, "days").days;
  const percent = Math.max(0, Math.min(100, Math.round(elapsed / total * 100)));
  return { percent, detail: `${start.toFormat("LL-dd")} 至 ${due.toFormat("LL-dd")} · ${Math.max(0, Math.round(total - elapsed))} 天余量` };
}

function normalizeMatchText(value) {

}

function progressValue(value) {
  if (value === null || value === undefined || value === "") return null;
  const match = String(value).match(/(\d+(?:\.\d+)?)\s*%?/);
  if (!match) return null;
  return Math.max(0, Math.min(100, Math.round(Number(match[1]))));
}

function progressFromText(text) {
  const raw = String(text ?? "");
  const match = raw.match(/(?:progress|进度|完成度)\s*(?::+|：|=)?\s*(\d+(?:\.\d+)?)\s*%?/i);
  return match ? progressValue(match[1]) : null;
}

function pageProgress(page, task) {
  const frontmatter = progressValue(page.progress ?? page["进度"] ?? page["完成度"] ?? page.percent ?? page["百分比"]);
  if (frontmatter !== null) return frontmatter;

  const taskName = cleanText(task.text);
  const tags = taskTags(task);
  const lines = arr(page.file?.lists)
    .map(item => String(item.text ?? "").trim())
    .filter(Boolean);
  const relatedLines = lines.filter(text => {
    return textMatchesTask(text, task);
  });

  for (const text of [...relatedLines, ...lines]) {
    const value = progressFromText(text);
    if (value !== null) return value;
  }

  return null;
}

      return {
        date: moment,
        type,
        cls,
        detail: detail || "手动补充说明",
        progress,
        path: ""
      };

    .filter(Boolean);
}

function makeTimelineEvents(task) {
  const events = [];
  const add = (dateValue, type, cls, detail = "", progress = null) => {
    const parsed = dateValue?.isLuxonDateTime ? dateValue : asTimelineMoment(dateValue);
    if (!parsed) return;
    events.push({ date: parsed, type, cls, detail, progress, path: notePathForDay(parsed) });
  };

  add(createdDay(task), "创建", "created", "任务创建日");
  add(startDay(task), "开始", "start", "任务进入可执行状态");
  add(scheduledDay(task), "计划", "plan", "计划处理日");
  add(dueDay(task), "截止", "due", "任务截止日");
  add(completedDay(task), "完成", "done", "任务完成记录");
  events.push(...timelineNotes(task));

  const taskName = cleanText(task.text);
  const tags = taskTags(task);
  const minDate = [createdDay(task), startDay(task), scheduledDay(task), dueDay(task), completedDay(task), today.minus({ days: 7 })]
    .filter(Boolean)
    .sort((a, b) => a.toMillis() - b.toMillis())[0] ?? today.minus({ days: 7 });
  const maxDate = [dueDay(task), completedDay(task), today.plus({ days: 7 })]
    .filter(Boolean)
    .sort((a, b) => b.toMillis() - a.toMillis())[0] ?? today.plus({ days: 7 });

  for (const page of allPages) {
    if (taskSourcePaths.includes(page.file.path)) continue;
    const matchedByProperty = pageMatchesTask(page, task);
    const day = pageTimelineDate(page, task);
    if (!day || day < minDate.minus({ days: 1 }) || day > maxDate.plus({ days: 1 })) continue;
    const snippets = listSnippets(page, 12).filter(text => {
      return textMatchesTask(text, task);
    });
    if (snippets.length || matchedByProperty) {
      const progress = pageProgress(page, task);
      events.push({
        date: day,
        type: matchedByProperty ? "文档" : "日报",
        cls: "report",
        detail: snippets.slice(0, 2).join(" / ") || page.file.name,
        progress,
        path: matchedByProperty ? "" : page.file.path,
        docPath: page.file.path
      });
    }
  }

  const sorted = events
    .sort((a, b) => a.date.toMillis() - b.date.toMillis())
    .filter((event, index, list) => {
      const prev = list[index - 1];
      return !prev || !(prev.date.toMillis() === event.date.toMillis() && prev.type === event.type && prev.detail === event.detail);
    });

  let lastProgress = null;
  return sorted.map(event => {
    const delta = event.progress !== null && event.progress !== undefined && lastProgress !== null
      ? event.progress - lastProgress
      : null;
    if (event.progress !== null && event.progress !== undefined) lastProgress = event.progress;
    return { ...event, delta };
  });
}

function statusControlsHtml(task) {

  `;
}

function timelineDraftKey(task) {
  return `tcc:timelineDraft:${taskId(task)}`;
}

function bindTimelineDraft(form, task) {
  const key = timelineDraftKey(task);
  const fields = ["moment", "type", "progress", "text"];
  const saved = (() => {
    try { return JSON.parse(window.localStorage.getItem(key) ?? "{}"); }
    catch { return {}; }
  })();

  for (const name of fields) {
    if (saved[name] !== undefined && form.elements[name]) {
      form.elements[name].value = saved[name];
    }
  }

  const save = () => {
    const draft = {};
    for (const name of fields) draft[name] = form.elements[name]?.value ?? "";
    window.localStorage.setItem(key, JSON.stringify(draft));
  };

  form.addEventListener("input", save);
  form.addEventListener("change", save);
}

function clearTimelineDraft(task) {
  window.localStorage.removeItem(timelineDraftKey(task));
}
